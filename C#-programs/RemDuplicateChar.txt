//remove duplicate
using System;
using System.Text;
class HelloWorld {
  static void Main() {
      string str = "excommunicadoyouuuuubwbwhwh"; //sky
      string newStr = RemoveDuplicate(str);
      Console.WriteLine(newStr+" ");
  }
 
  public static string RemoveDuplicate(string str)
  {
      StringBuilder sb = new StringBuilder();
      for(int i=0;i<str.Length;i++)
      {
          char ch = str[i];
          int j = 0;
          int count = 0;
          for(j=i+1;j<str.Length;j++)
          {
            if(str[j] == ch)
            {
                count++;
            }
          }
          if(count==0)
          {
              sb.Append(ch);
          }
      }
      return sb.ToString();
  }
}
