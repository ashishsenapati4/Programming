//reverse words in a string
using System;
using System.Text;
class HelloWorld {
  static void Main() {
      string str = "Tomorrow is a holiday"; //senapati Ashish
      string revString = ReverseEachWord(str);
      Console.WriteLine(revString+" ");
  }
 
  public static string ReverseEachWord(string str)
  {
      StringBuilder sb = new StringBuilder();
      int len = str.Length;
      for(int i=str.Length-1;i>=0;i--)
      {
          if(str[i] == ' ')
          {  
              int j = i+1;
              while(j<len)
              {
                  sb.Append(str[j++]);
              }
              sb.Append(' ');
              len = i;
          }
         
      }
      for(int i=0;i<len;i++)
      {
          sb.Append(str[i]);
      }
      return sb.ToString();
  }
}
