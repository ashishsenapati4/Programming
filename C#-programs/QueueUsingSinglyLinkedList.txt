using System;
public class HelloWorld
{
    public static void Main(string[] args)
    {
        QueueOperations qu = new QueueOperations();
        qu.Insert(90);
        qu.Insert(40);
        qu.Insert(78);
        qu.Insert(89);
        
        qu.Delete();
    }
}

public class Node
{
    public int data;
    public Node next;
    public Node(int _data)
    {
        this.data = _data;
    }
}
public class QueueOperations
{
    private Node head;
    private Node tail;
    
    public void Insert(int data)
    {
        Node newNode = new Node(data);
        if(head == null)
        {
            head = tail = newNode;
            return;
        }
        tail.next = newNode;
        tail = newNode;
        PrintElements(head);
        Console.WriteLine();
    }
    
   public Node Delete()
   {
       //removal from the head
       if(head == null)
       {
           Console.WriteLine("Nothing to remove");
           return head;
       }
       Node remNode = head;
       head = head.next;
       Console.WriteLine(remNode.data+" Deleted");
       PrintElements(head);
       return remNode;
   }
    
    public void PrintElements(Node node)
    {
        
        Console.WriteLine("Printing elements");
        while(node!=null)
        {
            Console.Write(node.data+" ");
            node = node.next;
        }
    }
}
