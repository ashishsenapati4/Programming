
using System;
using System.Collections.Generic;
class HelloWorld {
  static void Main() {
      int[] arr = {10, 5, 10};
      int k = GetSecondLargest(arr);
      Console.WriteLine(k+" ");
  }
  
  public static int GetSecondLargest(int[] arr)
  {
      Array.Sort(arr);
      
      for(int i=arr.Length-1;i>0;i--)
      {
          if(arr[i] != arr[i-1])
          {
            return arr[i-1];
          }
      }
      
      return - 1;
  }
  
}
